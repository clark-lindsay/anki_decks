try do
  do_something_that_may_fail(some_arg)
rescue
  ArgumentError ->
    IO.puts("Invalid argument given")
catch
  value ->
    IO.puts("Caught #{inspect(value)}")
else
  value ->
    IO.puts("Success! The result was #{inspect(value)}")
after
  IO.puts("This is printed regardless if it failed or succeeded")
end

The {{c1::rescue}} clause is used to {{c2::handle exceptions}} while the {{c1::catch}} clause can be used to {{c2::catch thrown values and exits}}. The {{c1::else}} clause can be used to {{c2::control flow based on the result of the expression}}. {{c1::catch}}, {{c1::rescue}}, and {{c1::else}} clauses work based on pattern matching (similar to the case special form).
